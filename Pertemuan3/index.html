# 1. Definisi struktur Node class Node: def __init__(self, data): self.data = data self.next = None # 2. Definisi Struktur Linked List class LinkedList: def __init__(self): self.head = None # Fungsi untuk menambah data di depan (agar mudah testing) def push(self, new_data): new_node = Node(new_data) new_node.next = self.head self.head = new_node # FUNGSI UTAMA: Menghapus node berdasarkan nilai def delete_node(self, key): temp = self.head # Jika head adalah node yang akan dihapus if temp and temp.data == key: self.head = temp.next temp = None return # Mencari node yang akan dihapus sambil melacak node sebelumnya prev = None while temp and temp.data != key: prev = temp temp = temp.next # Jika nilai tidak ditemukan if temp is None: print(f"Nilai {key} tidak ditemukan dalam list.") return # Memutuskan hubungan node yang dihapus prev.next = temp.next temp = None # Fungsi untuk mencetak isi list def print_list(self): temp = self.head while temp: print(temp.data, end=" -> ") temp = temp.next print("None") # --- CARA PENGGUNAAN --- llist = LinkedList() llist.push(10) llist.push(20) llist.push(30) llist.push(40) print("Linked List awal:") llist.print_list() # Menghapus angka 20 print("\nMenghapus 20...") llist.delete_node(20) llist.print_list() # Menghapus head (40) print("\nMenghapus 40 (Head)...") llist.delete_node(40) llist.print_list()
